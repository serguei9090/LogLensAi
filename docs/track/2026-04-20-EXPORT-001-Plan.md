**Goal:** Allow users to export filtered log results from the workspace to local CSV or JSON files via a native OS save dialog.

**Architecture:** 
1. **Backend**: Extend `api.py` with a new `export_logs` JSON-RPC method that queries DuckDB and writes the result to a specified file path.

3. **Integration**: Use Tauri's `save` dialog to let the user choose the destination and format, then call the sidecar to perform the heavy lifting.

```python
    async def method_export_logs(self, params: ExportLogsRequest):
        import csv
        import json
        
        # Reuse query_logs logic to get ALL matching logs (no limit)
        # Note: We might want to stream this if there are millions of logs, 
        # but for MVP, fetching into memory is acceptable.
        result = self.db.query_logs(
            workspace_id=params.workspace_id,
            query=params.query,
            filters=params.filters,
            limit=1000000, # Large limit for export
            offset=0,
            source_ids=params.source_ids,
            start_time=params.start_time,
            end_time=params.end_time,
        )
        logs = result.get("logs", [])
        
        if params.format == "csv":
            if not logs:
                with open(params.filepath, "w", newline="") as f:
                    f.write("")
                return {"status": "ok", "count": 0}
                
            keys = logs[0].keys()
            with open(params.filepath, "w", newline="", encoding="utf-8") as f:
                dict_writer = csv.DictWriter(f, fieldnames=keys)
                dict_writer.writeheader()
                dict_writer.writerows(logs)
        else:
            with open(params.filepath, "w", encoding="utf-8") as f:
                json.dump(logs, f, indent=2)
                
        return {"status": "ok", "count": len(logs)}
```

```tsx
  const handleExport = async () => {
    if (!activeWorkspaceId) return;

    try {
      const { save } = await import("@tauri-apps/plugin-dialog");
      const path = await save({
        title: "Export Logs",
        filters: [
          { name: "CSV", extensions: ["csv"] },
          { name: "JSON", extensions: ["json"] },
        ],
        defaultPath: `loglens_export_${new Date().toISOString().split("T")[0]}.csv`,
      });

      if (!path) return;

      const format = path.endsWith(".json") ? "json" : "csv";
      
      toast.loading(`Exporting logs to ${format.toUpperCase()}...`, { id: "export" });

      await callSidecar({
        method: "export_logs",
        params: {
          workspace_id: activeWorkspaceId,
          filepath: path,
          format,
          query: searchQuery || undefined,
          filters: filters.length > 0 ? filters : undefined,
          start_time: timeRange.start || undefined,
          end_time: timeRange.end || undefined,
        },
      });

      toast.success("Export complete", { id: "export", description: `Saved to ${path}` });
    } catch (e) {
      console.error("Export failed", e);
      toast.error("Export failed", { id: "export" });
    }
  };
```
